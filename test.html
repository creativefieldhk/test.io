        // --- 遊戲狀態與全域變數 ---

        // 資源庫

        const TRASH_EMOJIS = ['🥤', '🛍️', '🥫', '🗑️', '🗞️', '📦', '🛢️', '🧴'];

        const FISH_EMOJIS = ['🐟', '🐠', '🐢', '🐙', '🦀', '🐬', '🐳', '🐡', '🦑', '🦈'];


        // --- 畫布大小調整 ---

        function resizeCanvas() {

            canvas.width = window.innerWidth;

            canvas.height = window.innerHeight;

        // --- 滑鼠與觸控事件 (Google Sites 備用模式) ---

        function updateMousePos(evt) {

            const rect = canvas.getBoundingClientRect();

            let clientX, clientY;

            

            if (evt.touches && evt.touches.length > 0) {

                clientX = evt.touches[0].clientX;

                clientY = evt.touches[0].clientY;

            } else {

                clientX = evt.clientX;

                clientY = evt.clientY;

            }

            

            mousePosition = {

                x: clientX - rect.left,

                y: clientY - rect.top

            };

        }


        canvas.addEventListener('mousemove', updateMousePos);

        canvas.addEventListener('touchmove', (e) => { e.preventDefault(); updateMousePos(e); }, { passive: false });

        canvas.addEventListener('touchstart', (e) => { e.preventDefault(); updateMousePos(e); }, { passive: false });

        canvas.addEventListener('touchend', () => { mousePosition = null; });

        canvas.addEventListener('mouseleave', () => { mousePosition = null; });


        // --- UI 畫面切換控制 ---

        function switchScreen(screenId) {

            const screens = ['screen-init', 'screen-loading', 'screen-menu', 'screen-countdown', 'screen-gameover', 'game-hud'];

            screens.forEach(id => {

                document.getElementById(id).classList.add('hidden-screen');

            });

            if(screenId) {

                document.getElementById(screenId).classList.remove('hidden-screen');

            }

        }


        // --- 初始化 Mediapipe 手部追蹤 ---

        const hands = new Hands({locateFile: (file) => {

            return `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`;

        }});

        hands.setOptions({

            maxNumHands: 2,           

            modelComplexity: 0,       

            minDetectionConfidence: 0.6,

            minTrackingConfidence: 0.6

        });


        hands.onResults((results) => {

            handCursors = [];

            if (results.multiHandLandmarks) {

                for (const landmarks of results.multiHandLandmarks) {

                    const indexTip = landmarks[8];

                    handCursors.push({

                        x: (1 - indexTip.x) * canvas.width,

                        y: indexTip.y * canvas.height

                    });

                }

            }

        });


        async function processVideoFrame() {

            if (useCamera && video.readyState >= 2 && !video.paused) {

                await hands.send({image: video});

            }

            if (useCamera) {

                requestAnimationFrame(processVideoFrame);

            }

        }


        // --- 啟動遊戲流程 ---

        

        // 1A. 使用相機模式

        document.getElementById('btn-init-camera').addEventListener('click', async () => {

            switchScreen('screen-loading');

            try {

                const stream = await navigator.mediaDevices.getUserMedia({ 

                    video: { facingMode: 'user', width: { ideal: 640 }, height: { ideal: 480 } } 

                });

                video.srcObject = stream;

                await video.play();

                await hands.initialize();

                

                useCamera = true;

                processVideoFrame();

                

                gameState = 'menu';

                switchScreen('screen-menu');

                requestAnimationFrame(gameLoop);

            } catch (error) {

                console.error(error);

                alert('無法存取相機權限！如果您在 Google Sites 內，請改用「滑鼠/觸控模式」。系統已自動為您切換。');

                // 自動降級為觸控模式

                startMouseMode();

            }

        });


        // 1B. 使用滑鼠/觸控模式 (Google Sites 解法)

        document.getElementById('btn-init-mouse').addEventListener('click', startMouseMode);


        function startMouseMode() {

            useCamera = false;

            gameState = 'menu';

            switchScreen('screen-menu');

            // 直接啟動遊戲繪圖迴圈

            requestAnimationFrame(gameLoop);

        }


        // 2. 倒數與開始

        document.getElementById('btn-start').addEventListener('click', startCountdown);

        document.getElementById('btn-restart').addEventListener('click', startCountdown);


        function startCountdown() {

            gameState = 'countdown';

            switchScreen('screen-countdown');

            

            entities = [];

            particles = [];

            score = 0;

            timeLeft = 60;

            updateHUD();


            let count = 3;

            const countText = document.getElementById('countdown-text');

            countText.innerText = count;


            const countInterval = setInterval(() => {

                count--;

                if (count > 0) {

                    countText.innerText = count;

                    countText.style.animation = 'none';

                    countText.offsetHeight; 

                    countText.style.animation = null;

                } else {

                    clearInterval(countInterval);

                    startGame();

                }

            }, 1000);

        }


        // 3. 正式開始遊戲

        function startGame() {

            gameState = 'playing';

            switchScreen('game-hud');

            lastTime = performance.now();

            

            gameTimerInterval = setInterval(() => {

                if(gameState === 'playing') {

                    timeLeft--;

                    updateHUD();

                    if(timeLeft <= 0) endGame();

                }

            }, 1000);

        }


        function endGame() {

            gameState = 'gameover';

            clearInterval(gameTimerInterval);

            document.getElementById('final-score').innerText = score;

            switchScreen('screen-gameover');

        }


        function updateHUD() {

            document.getElementById('score-display').innerText = score;

            document.getElementById('timer-display').innerText = timeLeft;

        }


        // --- 遊戲邏輯與渲染 ---


        // 生成實體 (加入快慢速度變化與波浪參數)

        function spawnEntity() {

            const isTrash = Math.random() > 0.45; // 55% 是垃圾

            const type = isTrash ? 'trash' : 'fish';

            const emoji = isTrash ? 

                TRASH_EMOJIS[Math.floor(Math.random() * TRASH_EMOJIS.length)] : 

                FISH_EMOJIS[Math.floor(Math.random() * FISH_EMOJIS.length)];

            

            const size = 60 + Math.random() * 40; 

            let x, y, vx, vy;

            

            // 隨機速度倍率 (產生有快有慢的效果)

            const speedMult = 0.5 + Math.random() * 2.5; // 0.5x (很慢) 到 3.0x (極快)


            // 波浪移動的參數 (僅海洋生物使用)

            let baseY = 0;

            let waveSpeed = 0;

            let waveAmp = 0;

            let timeOffset = Math.random() * 100;


            if (isTrash) {

                // 垃圾掉落

                x = Math.random() * (canvas.width - 100) + 50;

                y = -100;

                vx = (Math.random() - 0.5) * 1.5; 

                vy = (2 + Math.random() * 3) * speedMult; // 掉落速度有快有慢

            } else {

                // 海洋生物從兩側游

                const fromLeft = Math.random() > 0.5;

                y = Math.random() * (canvas.height - 300) + 150;

                baseY = y;

                waveSpeed = 0.05 + Math.random() * 0.05; // 波浪頻率

                waveAmp = 15 + Math.random() * 35;       // 波浪幅度


                if (fromLeft) {

                    x = -100;

                    vx = (2 + Math.random() * 4) * speedMult; // 游動速度有快有慢

                } else {

                    x = canvas.width + 100;

                    vx = -(2 + Math.random() * 4) * speedMult;

                }

                vy = 0; 

            }


            entities.push({ 

                type, emoji, x, y, vx, vy, size, 

                baseY, waveSpeed, waveAmp, timeOffset, 

                lifeTime: 0 

            });

        }


        function spawnParticle(x, y, text, color) {

            particles.push({ x: x, y: y, text: text, color: color, life: 1.0 });

        }


        // 主迴圈

        function gameLoop(timestamp) {

            const deltaTime = timestamp - lastTime;

            lastTime = timestamp;


            // 1. 繪製背景

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            

            if (useCamera && video.readyState >= 2) {

                ctx.save();

                ctx.scale(-1, 1);

                const vRatio = video.videoWidth / video.videoHeight;

                const cRatio = canvas.width / canvas.height;

                let dW, dH, oX, oY;

                

                if (vRatio > cRatio) {

                    dH = canvas.height; dW = canvas.height * vRatio;

                    oX = (canvas.width - dW) / 2; oY = 0;

                } else {

                    dW = canvas.width; dH = canvas.width / vRatio;

                    oX = 0; oY = (canvas.height - dH) / 2;

                }

                ctx.drawImage(video, -oX - dW, oY, dW, dH);

                ctx.restore();

                ctx.fillStyle = 'rgba(10, 30, 80, 0.5)'; // 深海濾鏡

            } else {

                // 如果沒有相機，顯示深藍色漸層背景

                let grad = ctx.createLinearGradient(0, 0, 0, canvas.height);

                grad.addColorStop(0, '#0f172a');

                grad.addColorStop(1, '#1e3a8a');

                ctx.fillStyle = grad;

            }

            ctx.fillRect(0, 0, canvas.width, canvas.height);


            // 2. 背景氣泡動畫

            if(Math.random() < 0.1) {

                bubbles.push({

                    x: Math.random() * canvas.width, y: canvas.height + 20,

                    size: Math.random() * 6 + 2, speed: Math.random() * 3 + 1

                });

            }

            ctx.fillStyle = 'rgba(255, 255, 255, 0.2)';

            for(let i = bubbles.length - 1; i >= 0; i--) {

                let b = bubbles[i];

                b.y -= b.speed;

                ctx.beginPath();

                ctx.arc(b.x, b.y, b.size, 0, Math.PI*2);

                ctx.fill();

                if(b.y < -50) bubbles.splice(i, 1);

            }


            // 整合所有目前的游標 (手部AI + 備用滑鼠/觸控)

            let activeCursors = [...handCursors];

            if (mousePosition) {

                activeCursors.push(mousePosition);

            }


            // 3. 遊戲進行邏輯

            if (gameState === 'playing') {

                spawnTimer -= deltaTime;

                if (spawnTimer <= 0) {

                    spawnEntity();

                    // 生成間隔加快，讓畫面更豐富

                    spawnTimer = 400 + Math.random() * 600; 

                }


                for (let i = entities.length - 1; i >= 0; i--) {

                    let e = entities[i];

                    e.lifeTime++;

                    e.x += e.vx;

                    

                    // 海洋生物加上波浪形軌跡，垃圾維持直線掉落

                    if (e.type === 'fish') {

                        e.y = e.baseY + Math.sin((e.lifeTime + e.timeOffset) * e.waveSpeed) * e.waveAmp;

                    } else {

                        e.y += e.vy;

                    }


                    ctx.save();

                    ctx.translate(e.x, e.y);

                    

                    // 繪製顯明顏色的光圈/底圖

                    ctx.beginPath();

                    ctx.arc(0, 0, e.size * 0.55, 0, Math.PI * 2);

                    if (e.type === 'trash') {

                        // 垃圾：綠色光圈 (提示玩家捕捉)

                        ctx.fillStyle = 'rgba(74, 222, 128, 0.5)';

                        ctx.shadowColor = '#4ade80';

                        ctx.shadowBlur = 20;

                    } else {

                        // 海洋生物：紅色光圈 (提示危險)

                        ctx.fillStyle = 'rgba(248, 113, 113, 0.5)';

                        ctx.shadowColor = '#f87171';

                        ctx.shadowBlur = 20;

                    }

                    ctx.fill();

                    ctx.shadowBlur = 0; // 重置陰影以免影響Emoji


                    // 魚往左游時翻轉圖片

                    if (e.vx < 0 && e.type === 'fish') {

                        ctx.scale(-1, 1);

                    }

                    

                    ctx.font = `${e.size}px sans-serif`;

                    ctx.textAlign = 'center';

                    ctx.textBaseline = 'middle';

                    ctx.fillText(e.emoji, 0, 0);

                    ctx.restore();


                    // 碰撞偵測

                    let hit = false;

                    for (let cursor of activeCursors) {

                        const dist = Math.hypot(e.x - cursor.x, e.y - cursor.y);

                        const hitRadius = (e.size / 2) + 30; // 稍微放寬判定


                        if (dist < hitRadius) {

                            hit = true;

                            if (e.type === 'trash') {

                                score += 3;

                                spawnParticle(e.x, e.y, '+3', '#4ade80');

                            } else {

                                score -= 1;

                                spawnParticle(e.x, e.y, '-1', '#f87171');

                            }

                            updateHUD();

                            break; 

                        }

                    }


                    if (hit || e.y > canvas.height + 150 || e.x < -150 || e.x > canvas.width + 150) {

                        entities.splice(i, 1);

                    }

                }

            }


            // 4. 繪製互動游標 (手部或滑鼠)

            for (let cursor of activeCursors) {

                ctx.beginPath();

                ctx.arc(cursor.x, cursor.y, 25, 0, Math.PI * 2);

                ctx.fillStyle = 'rgba(56, 189, 248, 0.4)'; 

                ctx.fill();

                

                ctx.beginPath();

                ctx.arc(cursor.x, cursor.y, 10, 0, Math.PI * 2);

                ctx.fillStyle = '#fff';

                ctx.fill();

                ctx.strokeStyle = '#0284c7';

                ctx.lineWidth = 3;

                ctx.stroke();

            }


            // 5. 繪製得分/扣分特效

            for (let i = particles.length - 1; i >= 0; i--) {

                let p = particles[i];

                p.life -= deltaTime / 1000;

                p.y -= 3; 


                if (p.life <= 0) {

                    particles.splice(i, 1);

                } else {

                    ctx.globalAlpha = p.life;

                    ctx.font = '900 45px sans-serif';

                    ctx.textAlign = 'center';

                    ctx.textBaseline = 'middle';

                    

                    ctx.strokeStyle = '#fff';

                    ctx.lineWidth = 8;

                    ctx.strokeText(p.text, p.x, p.y);

                    

                    ctx.fillStyle = p.color;

                    ctx.fillText(p.text, p.x, p.y);

                    ctx.globalAlpha = 1.0;

                }

            }


            requestAnimationFrame(gameLoop);

        }


    </script>
